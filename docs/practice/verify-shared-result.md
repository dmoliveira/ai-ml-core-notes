# Verify Shared Result

Use this helper to verify a shared quiz link signature before trusting its content.

1. Paste a full shared URL generated by the quiz page.
2. Click **Verify**.
3. Check diagnostics (signature, parameters, payload size).
4. Copy decoded JSON if needed.

## Verification incident loop

1. Verify the shared URL and inspect diagnostics.
2. If signature is invalid, treat the payload as untrusted.
3. Compare expected/provided signatures and confirm source URL.
4. Re-generate and re-verify before sharing results externally.

const SHARE_SIGNATURE_SALT = "aiml-quiz-share-v1";

function computeShareSignature(payloadJson) {
  const combined = `${payloadJson}|${SHARE_SIGNATURE_SALT}`;
  let hash = 0;
  for (let idx = 0; idx < combined.length; idx += 1) {
    hash = (hash << 5) - hash + combined.charCodeAt(idx);
    hash |= 0;
  }
  return Math.abs(hash).toString(36);
}

function decodePayload(encoded) {
  try {
    const json = decodeURIComponent(escape(atob(encoded)));
    return { json, payload: JSON.parse(json) };
  } catch {
    return null;
  }
}

function setDiagnostics(lines) {
  const target = document.getElementById("verify-share-diagnostics");
  target.textContent = lines.join(" | ");
}

function verifyRawUrl(rawUrl) {
  const feedback = document.getElementById("verify-share-feedback");
  const output = document.getElementById("verify-share-output");

  if (!rawUrl) {
    feedback.textContent = "Paste a shared URL first.";
    output.textContent = "";
    setDiagnostics(["url: missing"]);
    return;
  }

  let parsed;
  try {
    parsed = new URL(rawUrl);
  } catch {
    feedback.textContent = "Invalid URL format.";
    output.textContent = "";
    setDiagnostics(["url: invalid"]);
    return;
  }

  const encoded = parsed.searchParams.get("result");
  const signature = parsed.searchParams.get("sig");
  if (!encoded || !signature) {
    feedback.textContent = "URL is missing result or sig parameters.";
    output.textContent = "";
    setDiagnostics([
      `result param: ${encoded ? "ok" : "missing"}`,
      `sig param: ${signature ? "ok" : "missing"}`,
    ]);
    return;
  }

  const decoded = decodePayload(encoded);
  if (!decoded) {
    feedback.textContent = "Unable to decode result payload.";
    output.textContent = "";
    setDiagnostics(["payload: decode failed"]);
    return;
  }

  const expected = computeShareSignature(decoded.json);
  const valid = expected === signature;
  feedback.textContent = valid ? "Signature valid." : "Signature invalid.";
  output.textContent = JSON.stringify(decoded.payload, null, 2);
  setDiagnostics([
    `result param: ok`,
    `sig param: ok`,
    `expected sig: ${expected}`,
    `provided sig: ${signature}`,
    `payload chars: ${decoded.json.length}`,
  ]);
}

async function copyPayload() {
  const text = document.getElementById("verify-share-output").textContent || "";
  const feedback = document.getElementById("verify-share-feedback");
  if (!text) {
    feedback.textContent = "Nothing to copy yet.";
    return;
  }
  try {
    await navigator.clipboard.writeText(text);
    feedback.textContent = "Decoded payload copied.";
  } catch {
    feedback.textContent = "Unable to copy automatically.";
  }
}

function clearVerifier() {
  document.getElementById("verify-share-url").value = "";
  document.getElementById("verify-share-feedback").textContent = "";
  document.getElementById("verify-share-diagnostics").textContent = "";
  document.getElementById("verify-share-output").textContent = "";
}

document.getElementById("verify-share-button").addEventListener("click", () => {
  const rawUrl = String(document.getElementById("verify-share-url").value || "").trim();
  verifyRawUrl(rawUrl);
});

document.getElementById("verify-share-copy").addEventListener("click", copyPayload);
document.getElementById("verify-share-clear").addEventListener("click", clearVerifier);
document.getElementById("verify-share-load-current").addEventListener("click", () => {
  document.getElementById("verify-share-url").value = window.location.href;
});
</script>
